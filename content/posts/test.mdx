# 读取文本文件并统计单词频率
from collections import Counter
import re

def word_frequency(filename):
    with open(filename, 'r', encoding='utf-8') as file:
        text = file.read().lower()
        words = re.findall(r'\b\w+\b', text)
        return Counter(words)
